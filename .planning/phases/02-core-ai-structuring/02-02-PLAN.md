```typescript
import { z } from 'zod'

export const FiveLayerModelSchema = z.object({
  goal: z.object({ summary: z.string(), before: z.string(), after: z.string(), metrics: z.array(z.string()) }),
  assumption: z.object({ items: z.array(z.object({ content: z.string(), confidence: z.enum(['high','medium','low']), rationale: z.string() })) }),
  behavior: z.object({ actors: z.array(z.string()), actions: z.array(z.object({ actor: z.string(), action: z.string(), precondition: z.string().optional(), postcondition: z.string().optional() })) }),
  scenario: z.object({ normal: z.array(...), edge: z.array(...), error: z.array(...) }),
  verifiability: z.object({ automated: z.array(...), manual: z.array(...) }),
})
export type FiveLayerModel = z.infer<typeof FiveLayerModelSchema>
```

  <name>Task 1: Install AI SDK and create structuring engine with retry</name>
  <files>src/server/ai/prompt.ts, src/server/ai/structuring.ts, .env.example</files>
  <action>
1. Install packages: `npm install ai @ai-sdk/openai`

2. Add `OPENAI_API_KEY=` to `.env.example` (append, don't overwrite existing lines).

3. Create `src/server/ai/prompt.ts`:
   - Export `buildStructuringPrompt(userInput: string): string`
   - System prompt in English instructing the LLM to:
     - Analyze natural language requirement into five-layer model
     - Describe each layer's purpose (Goal/Assumption/Behavior/Scenario/Verifiability)
     - Match output language to input language
     - Be specific and actionable, not generic
     - Surface hidden assumptions the user hasn't considered
     - Generate at least 2 scenarios per category (normal/edge/error)
   - Append the user's raw input at the end

4. Create `src/server/ai/structuring.ts`:
   - Import `generateText`, `Output` from `ai`
   - Import `openai` from `@ai-sdk/openai`
   - Import `FiveLayerModelSchema` from `@/lib/schemas/requirement`
   - Import `eventBus` from `@/server/events/bus`
   - Import `buildStructuringPrompt` from `./prompt`

   - **Zod 4 compatibility check:** First try `Output.object({ schema: FiveLayerModelSchema })` directly. If AI SDK 6 doesn't accept Zod 4 schemas natively, use the fallback bridge pattern:
     ```typescript
     import { jsonSchema } from 'ai'
     import { z } from 'zod'
     const aiSchema = jsonSchema(z.toJSONSchema(FiveLayerModelSchema))
     ```
     Use whichever approach works. Document the choice with a comment.

   - Export `generateStructuredModel(input: string, requirementId: string, userId: string)`:
     - MAX_RETRIES = 3
     - Emit `requirement.structuring.started` before loop
     - Loop up to 3 attempts:
       - Call `generateText({ model: openai('gpt-4o'), output: Output.object({ schema }), prompt: buildStructuringPrompt(input) })`
       - On success: emit `requirement.structuring.completed`, return `{ success: true, model: output, attempts }`
       - On error: if last attempt, emit `requirement.structuring.failed`, return `{ success: false, error, attempts }`
       - Otherwise: silent retry (continue loop)
     - Return type: `{ success: boolean; model?: FiveLayerModel; error?: unknown; attempts: number }`
  </action>
  <verify>
    <automated>npx tsc --noEmit</automated>
  </verify>
  <done>AI SDK installed. buildStructuringPrompt returns a bilingual system prompt. generateStructuredModel implements 3-attempt retry with event bus emissions. Zod 4 compatibility resolved (native or bridge). TypeScript compiles.</done>

  <files>src/app/api/ai/structure/route.ts</files>

1. Create `src/app/api/ai/structure/route.ts`:
   - This is a Next.js App Router API route (NOT tRPC — useObject hook needs a standard HTTP endpoint)
   - Import `streamText`, `Output` from `ai`
   - Import `openai` from `@ai-sdk/openai`
   - Import `FiveLayerModelSchema` from `@/lib/schemas/requirement`
   - Import `buildStructuringPrompt` from `@/server/ai/prompt`
   - Import `verifySession` from `@/lib/dal` (per established codebase pattern — verifySession is the cached auth check used in all non-tRPC server code; do NOT use getSession from @/server/auth/session here)
   - Import `eventBus` from `@/server/events/bus`

   - Export async `POST(req: Request)`:
     - Call `const session = await verifySession()`, return 401 if not authenticated
     - Parse body: `{ rawInput: string, requirementId?: string }`
     - Validate rawInput is non-empty string, return 400 if invalid
     - Emit `requirement.structuring.started` if requirementId provided (use session.userId)
     - Call `streamText({ model: openai('gpt-4o'), output: Output.object({ schema: FiveLayerModelSchema or jsonSchema bridge }), prompt: buildStructuringPrompt(rawInput) })`
     - Return `result.toTextStreamResponse()` (the AI SDK streaming response)
     - Note: Streaming does NOT support retry — the retry loop in structuring.ts is for non-streaming server-side calls. The streaming route is best-effort single attempt. If the client needs retry, it re-submits.
  </action>
